# build strategy that encourages the AI to generally do things that are beneficial for when it or its territory gets annexed
HMM_ai_annex_build_infra_until_off_toaster_eco = {
    allowed = {
		OR = { # Don't apply this to "main" countries to prevent test build interference
            is_playable_country = no
            is_annexable_playable_country = yes
        }
	}
    enable = {
		OR = {
            has_idea = undisturbed_isolation
			has_idea = isolation
            has_idea = civilian_economy
        }
	}
    abort_when_not_enabled = yes
    ai_strategy = {
        type = building_target
        id = infrastructure
        value = 9999
    }
}

HMM_ai_annex_build_reserve_pp_for_partial = {
    allowed = {
		OR = { # Don't apply this to "main" countries to prevent test build interference
            is_playable_country = no
            is_annexable_playable_country = yes
        }
	}
    enable = {
        OR = {
            has_war_support > 0.249
            AND = { # these countries have special advisors that give +10% war support, so should start saving up for partial at lower WS
                has_war_support > 0.149
                OR = {
                    original_tag = SIA
                    original_tag = MAN
                }
            }
        }
		NOR = {
			has_idea = partial_economic_mobilisation
			has_idea = war_economy
			has_idea = tot_economic_mobilisation
		}
	}
    abort_when_not_enabled = yes
    ai_strategy = {
        type = pp_spend_amount
        id = idea
        value = 500
    }
}
