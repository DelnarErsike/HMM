on_actions = {
	on_startup = {
		effect = {
			hmm_intel = yes
			every_country = {
				limit = {
					is_playable_country = yes
				}
				add_ideas = {
					peace_training
				}
			}
			every_country = {
				every_army_leader = {
					limit = {
						NOT = { 
							has_trait = no_grind
						}
					}
				add_unit_leader_trait = no_grind
				}
			}
			every_country = {
				limit = {
					OR = {
						ai_annex = yes
						ai_rk = yes
						ai_manpower = yes
						ai_black = yes
					}
				}
				country_lock_all_division_template = yes
				add_ideas = {
					ai_tradelaw
				}
			}
			SOV = { 
				country_event = {
					id = mexico.21 # Mexico returns trotsky
					days = 396 # Feb 37
				}
			}
			every_country = {
				limit = {
					OR = {
						ai_annex = yes
						ai_black = yes
					}
				}
				add_ideas = {
					ai_annex
				}
			}
			every_country = {
				limit = {
					OR = {
						ai_manpower = yes
					}
				}
				add_ideas = {
					ai_war
				}
			}

			every_country = {
				limit = {
					AND = {
						is_playable_country = yes
						NOT = {
							tag = SPR
						}
					}

				}
				add_ideas = {
					novols
				}
			}
			SPR = {
				add_ideas = {
					no_faction
				}
			}
			every_country = {
				add_ideas = {
					Spies_no_capture
				}
			}
		}
	}
	on_monthly = {
		effect = {
			every_country = {
				limit = {
					is_playable_country = yes
					NOT = {
						OR = {
							has_war = yes
							has_idea = peace_training
							AND = {
								tag = SPR
								any_country = {
									has_volunteers_amount_from = {
										tag = SPR
										count > 0
									} 
								}	
							}
						}
					}
				}
				add_ideas = peace_training
			}
			if = {
				limit = {
					OR={
						BEL = {
							has_capitulated = no
						}
						HOL = {
							has_capitulated = no
						}
						LUX = {
							has_capitulated = no
						}
						DEN = {
							has_capitulated = no
						}
						NOR = {
							has_capitulated = no
						}
						POL = {
							has_capitulated = no
						}
					}
				}
				every_country = {
					limit = {
						axis = yes
					}
					every_army_leader = {
						limit = {
							NOT = { 
								has_trait = no_grind
							}
						}
						add_unit_leader_trait = no_grind
					}
				}
			}
			if = {
				limit = {
					GER = {has_war = no }
				}
				every_country = {
					limit = {
						allies = yes
					}
					every_army_leader = {
						limit = {
							NOT = { 
								has_trait = no_grind
							}
						}
						add_unit_leader_trait = no_grind
					}
				}
			}
		}
	}
	on_capitulation = {
		effect = {
			hmm_capitulations_on_actions = yes

			if = {
				limit = {
						BEL = {
							has_capitulated = no
						}
						HOL = {
							has_capitulated = no
						}
						LUX = {
							has_capitulated = no
						}
						DEN = {
							has_capitulated = no
						}
						NOR = {
							has_capitulated = no
						}
						POL = {
							has_capitulated = no
						}
				}
				every_country = {
					limit = {
						axis = yes
					}
					every_army_leader = {
						remove_unit_leader_trait = no_grind
					}
				}
			}
		}
	}
	on_state_control_changed = {
		effect = {
			if = {
				limit = {
					FROM = { tag = FRA }
					FROM.FROM = { is_core_of = FRA }
					OR ={  AND ={  FRA = { surrender_progress > 0.32 NOT = { controls_state = 16 } } }	FRA = { surrender_progress	> 0.5	}	 }
					NOT = { FRA = { has_country_flag = france_tp} }
				}


			FRA = { 
				set_country_flag = france_tp
				transfer_units_fraction = {
					target = COG
					size = 1
					stockpile_ratio = 1
					army_ratio = 0
					navy_ratio = 0
					air_ratio = 0
					keep_unit_leaders_trigger = {
						always = yes
						}
					}					
				}
			
			}
		}
	}
	on_war_relation_added = {
        effect = {
            if = {
                limit = {
                    tag = JAP
                    FROM = {
                        is_in_faction_with = ENG
                    }
                }
                every_country = {
                    limit = {
                        is_in_faction_with = ENG
                    }
                    add_to_war = {
                        targeted_alliance = FROM
                        enemy = JAP
                        hostility_reason = asked_to_join
                    }
                }
                every_country = {
                    limit = {
                        is_in_faction_with = JAP
                    }
                    add_to_war = {
                        targeted_alliance = ROOT
                        enemy = FROM
                        hostility_reason = asked_to_join
                    }
                }
            }
        }
    }
	on_declare_war = {
		effect = {
			if = {
				limit = {
					FROM = {
						tag = POL
					}
				}
				every_country = {
					limit = {
						allies = yes
					}
					every_army_leader = {
						remove_unit_leader_trait = no_grind
					}
				}
				ENG = {
					puppet = IRE
					set_autonomy = {
						target = IRE
						autonomy_state = autonomy_integrated_puppet
					}
				}
				

			}
			if = {
				limit = {
					FROM = {
						allies = yes
					}
				}
				ENG = {
					add_to_faction = FROM
				}
			}
			if = {
                limit = {
                    FROM = {
                        tag = ALB
                    }
                }
                ALB = {
                    load_oob = "ALB_CHROM_SPAWN"
                }
            }
            if = {
                limit = {
                    FROM = {
                        tag = BEL
                    }
                }
                BEL = {
                    load_oob = "BEL_CHROM_SPAWN"
                }
            }
            if = {
                limit = {
                    FROM = {
                        tag = DEN
                    }
                }
                DEN = {
                    load_oob = "DEN_CHROM_SPAWN"
                }
            }
            if = {
                limit = {
                    FROM = {
                        tag = HOL
                    }
                }
                HOL = {
                    load_oob = "HOL_CHROM_SPAWN"
                }
            }
            if = {
                limit = {
                    FROM = {
                        tag = LUX
                    }
                }
                LUX = {
                    load_oob = "LUX_CHROM_SPAWN"
                }
            }
            if = {
                limit = {
                    FROM = {
                        tag = NOR
                    }
                }
                NOR = {
                    load_oob = "NOR_CHROM_SPAWN"
                }
            }
			if = {
                limit = {
                    FROM = {
                        tag = POL
                    }
                }
                POL = {
                    load_oob = "POL_CHROM_SPAWN"
                }
            }
			if = {
                limit = {
                    FROM = {
                        tag = YUG
                    }
                }
                YUG = {
                    load_oob = "YUG_CHROM_SPAWN"
                }
            }
			if = {
                limit = {
                    FROM = {
                        tag = GRE
                    }
                }
                GRE = {
                    load_oob = "GRE_CHROM_SPAWN"
                }
            }
			if = {
                limit = {
                    FROM = {
                        tag = CHI
                    }
                }
                CHI = {
                    load_oob = "CHI_CHROM_SPAWN"
                }
            }
			
		}
	}
}
