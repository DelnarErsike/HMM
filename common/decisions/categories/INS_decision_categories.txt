INS_national_revolution_cat = {
	icon = GFX_decision_category_INS_national_revolution
	priority = 100
	visible_when_empty = yes

	scripted_gui = INS_revolution_scripted_gui

	on_map_area = {
		target_trigger = {
			FROM = {
				has_state_flag = INS_owned_states_at_game_start
				OWNER = {
					OR = {
						tag = HOL
						tag = event_target:INS_dutch_indonesia
					}
				}
				any_neighbor_state = {
					OR = {
						is_owned_by = INS
						has_dynamic_modifier = { modifier = INS_state_militia_activity_1 }
						has_dynamic_modifier = { modifier = INS_state_militia_activity_2 }
						has_dynamic_modifier = { modifier = INS_state_militia_activity_3 }
					}
				}
				any_neighbor_state = {
					OR = {
						is_owned_by = HOL
						is_owned_by = event_target:INS_dutch_indonesia
					}
					impassable = no
				}
				impassable = no
			}
		}
		name = INS_national_revolution_cat_map
		zoom = 600
	}

	allowed = {
		original_tag = INS
		NOT = { tag = event_target:INS_dutch_indonesia }
		has_dlc = "Thunder at Our Gates"
	}

	visible = {
		INS = {
			has_completed_focus = INS_the_declaration_of_independence
			has_country_flag = { flag = INS_revolution_has_started }
		}
	}
}
